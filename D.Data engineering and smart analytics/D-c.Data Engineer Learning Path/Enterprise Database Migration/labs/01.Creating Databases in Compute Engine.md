1. Create a MySQL database on Linux

```bash
sudo apt update
sudo apt install -y mysql-server

sudo mysql_secure_installation

sudo mysql -u root -p
```

```sql
SHOW databases;
CREATE database petsdb;
USE petsdb;
CREATE TABLE pets (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255),
    breed VARCHAR(255)
);

INSERT INTO pets (name, breed)
VALUES ('Noir', 'Schnoodle');

SELECT * FROM pets;
```
